rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {    
    match /publicEvents/{eventId} {
      allow read: if request.auth != null;
      allow write: if request.resource.data.hostId == request.auth.uid;
      
      match /usersJoined/{userId} {
        allow read: if request.auth != null;
        allow write: if userId == request.auth.uid;
    	}
    }  
    
    match /privateEvents/{eventId} {
      allow read: if request.auth.uid in request.resource.data.invitedUsers || request.resource.data.hostId == request.auth.uid;
      allow write: if request.resource.data.hostId == request.auth.uid;
      
      match /usersJoined/{userId} {
        allow read: if request.auth.uid in get(/databases/$(database)/documents/privateEvents/$(eventId)).data.invitedUsers || get(/databases/$(database)/documents/privateEvents/$(eventId)).data.hostId == request.auth.uid;
        allow write: if userId == request.auth.uid && (request.auth.uid in get(/databases/$(database)/documents/privateEvents/$(eventId)).data.invitedUsers || get(/databases/$(database)/documents/privateEvents/$(eventId)).data.hostId == request.auth.uid);
    	}
    }
    
    match /userProfiles/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth.uid == userId;
    }
  }
}
